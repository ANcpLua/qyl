## Orchestration Architecture

/// <summary>
///     The multi-agent workflow pattern for parallel optimization.
/// </summary>
/// <remarks>
///     <para>
///         This system uses an orchestrator-worker pattern where the lead agent (you)
///         coordinates specialized subagents that operate in parallel. Each subagent
///         has its own context window, enabling:
///         <list type="bullet">
///             <item><description>Parallel exploration of different optimization dimensions</description></item>
///             <item><description>Compression of findings back to the lead agent</description></item>
///             <item><description>Separation of concerns with distinct tools and prompts</description></item>
///             <item><description>Reduced path dependency through independent investigations</description></item>
///         </list>
///     </para>
///     <para>
///         <b>Token economics:</b> Multi-agent systems use ~15x more tokens than single chats,
///         but the performance gains justify the cost for valuable optimization tasks.
///         Token usage explains 80% of performance variance in complex tasks.
///     </para>
/// </remarks>


## Parallel Subagent Deployment

/// <summary>
///     How to spawn and coordinate parallel subagents for optimization tasks.
/// </summary>
/// <remarks>
///     <para>
///         Use the Task tool with <c>run_in_background: true</c> to launch multiple
///         subagents simultaneously. Each subagent receives a focused task and returns
///         compressed findings.
///     </para>
///     <example>
///         <b>Spawning the tribe for agent optimization:</b>
///         <code>
///         // Launch all subagents in parallel (single message, multiple Task calls)
///
///         Task 1: Explore subagent - "Analyze the agent's architecture and dependencies"
///         Task 2: Explore subagent - "Audit the agent's description and trigger conditions"
///         Task 3: Explore subagent - "Review token efficiency and progressive disclosure"
///         Task 4: Explore subagent - "Examine error handling and edge cases"
///         Task 5: Explore subagent - "Assess documentation clarity and completeness"
///
///         // All run simultaneously, each returns compressed findings
///         </code>
///     </example>
///     <para>
///         <b>Subagent task structure:</b> Each subagent needs:
///         <list type="number">
///             <item><description>Clear objective with specific deliverable</description></item>
///             <item><description>Output format specification</description></item>
///             <item><description>Guidance on tools and sources to use</description></item>
///             <item><description>Clear task boundaries (avoid duplication)</description></item>
///         </list>
///     </para>
/// </remarks>

///     Verification requirements for optimizing SKILL.md files.

/// <remarks>
///     <list type="bullet">
///         <item>
///             <description>

///             <description>

///             <description>
///                 <b>File references</b>: One level deep from SKILL.md (no nested references)

///             <description>
///                 <b>Time-sensitive info</b>: None, or relegated to "old patterns" section
